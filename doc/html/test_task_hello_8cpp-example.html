    // enregisters a test.
    class Test : testing::unit::Base
    {
    public :
        Test() : testing::unit::Base("task_hello") {}

    private :
        // tests task::distributor::Batch by printing strings concurrently.
        virtual bool run_impl();
    };
    Test const g_test;

    // a task to print "hello ${this->id_}".
    struct HelloTask
    {
        typedef task::distributor::Batch<HelloTask>
            distributor_type;
        
        typedef task::worker::Batch<HelloTask>
            worker_type;
        
        HelloTask() : id_(0) {}
        
        HelloTask(size_t id) : id_(id) {}

        void operator()(worker_type &);

        reduction_type reduce(reduction::name_type const & name) const;

        size_t id_;
    };

    reduction_type reduce(
        reduction::name_type const & name, HelloTask const & src);

    typedef HelloTask                       task_type;
    typedef task_type::worker_type          worker_type;
    typedef task_type::distributor_type     distributor_type;

}

/*
  @brief        tests task::distributor::Batch by printing strings concurrently.
  @since        2011-12-17T19:54:00+0900
 */
bool Test::run_impl()
{
    // initializes logging
    testing::Log log(*this);

    // initializes a random number generator
    testing::Random random;

    log.reduce().append("random", random);

    // temporary buffer for dynamic memory allocation
    size_t const    buffer_size = 0x400 * 0x400;
    size_t          buffer[buffer_size / sizeof(size_t)] = {0, };
    
    // initializes heap.
    typedef memory::heap::Arbitrary<>   heap_type;
    heap_type                           heap;
    if (!heap.reset(buffer, buffer_size)) {
        hryky_log_alert(
            (*reduction::map("failed to reset memory::heap::Arbitrary"))
            .append("buffer_size",      buffer_size));
        return false;
    }

    // decorates heap as thread-safe.
    typedef memory::heap::Lock<heap_type>   lockable_heap_type;
    lockable_heap_type                      lockable_heap(heap);

    // creates adapter to treat the heap polymorphically.
    typedef memory::heap::Adapter<lockable_heap_type>   adapter_heap_type;
    adapter_heap_type                                   adapter_heap(lockable_heap);

    // initializes a pool of thread resources.
    thread::Pool thread_pool;
    if (!thread_pool.reset(thread::Pool::ResetParameter())) {
        hryky_log_err(
            "failed to reset thread::Pool");
        return false;
    }
    
    // decides the number of concurrent units
    size_t const threads_size = 8;

    processor_id_type const preferred_processors[threads_size] =
    {
        0, 1, 2, 3, 4, 5, 6, 7,
    };

    // initializes the manager of concurrent sequences.
    task::Concurrency concurrency;
    if (!concurrency.reset(
        task::Concurrency::ResetParameter(
            adapter_heap,
            thread_pool)
        .sequences_size(threads_size)
        .preferred_processors(preferred_processors)))
    {
        hryky_log_emerg(
            (*reduction::map("failed to reset concurrent module"))
                .append("the number of threads", threads_size));
        return false;
    }

    // decides the number of tasks
    random::distribution::Normal<size_t> tasks_size_dist(random.generator(), 0x1000);
    (tasks_size_dist
     .mean(0x100)
     .min(1)
     .max(0x100000));
    
    size_t const tasks_size = tasks_size_dist();
    
    log.reduce().append("tasks_size", tasks_size).append("tasks_size_dist", tasks_size_dist);

    // decides the size of batch
    random::distribution::Normal<size_t>
        batch_size_dist(random.generator(), std::max<size_t>(1, tasks_size >> 2));
    (batch_size_dist
     .mean(tasks_size >> 1)
     .min(1)
     .max(tasks_size));

    size_t const batch_size = batch_size_dist();

    log.reduce().append("batch_size", batch_size).append("batch_size_dist", batch_size_dist);

    // initializes the distributor of tasks
    distributor_type distributor;
    if (!distributor.reset(
        distributor_type::ResetParameter(concurrency)
        .batch_size(batch_size)))
    {
        hryky_log_emerg(
            (*reduction::map("failed to reset a task distributor"))
            .append("the size of a batch of tasks", batch_size));
        return false;
    }

    // distributes tasks
    size_t task_id = 0;
    for (; tasks_size != task_id; ++task_id) {
        if (!distributor.distribute(task_type(task_id))) {
            hryky_log_alert(
                (*reduction::map("failed to append a task"))
                .append("task_id", task_id)
                );
            return false;
        }
    }

    // attaches distributor to concurrent units
    if (!distributor.attach()) {
        hryky_log_alert("failed to attach workers to concurrent units");
        return false;
    }

    // waits until the completion of tasks
    if (!distributor.join()) {
        hryky_log_alert(
            "failed to wait that the workers complete tasks");
        return false;
    }

    return true;
}
/*
  @brief        prints "hello this->id_".
  @since        2011-11-29T20:23:17+0900
 */
void HelloTask::operator()(worker_type &)
{
    hryky_log_debug(
        *hryky_reduce("hello", this->id_));
}
reduction_type HelloTask::reduce(reduction::name_type const & name) const {
    reduction::map_type map = reduction::map(name);

    map->append("id", this->id_);

    return map;
}
reduction_type reduce(
    reduction::name_type const & name, HelloTask const & src) {
    return src.reduce(name);
}
