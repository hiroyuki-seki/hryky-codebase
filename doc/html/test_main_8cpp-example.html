/*
  @brief        picks a test by command-line options.
  @since        2011-12-17T15:37:12+0900
 */
int hryky::testing::main_impl(int const argc, char const * const argv[])
{
    {
        /*
          @attention
          - boost::filesystem::path::codecvt() has a static variable which
            calls operator new. The corresponding operator delete is called
            at the end of program.
         */
        filesystem::path_type::codecvt();

        std::ostringstream os;
        os << static_cast<uint64_t>(0);

        std::locale locale;
        std::use_facet<std::codecvt<char,char,int> >(locale);
    }

    typedef memory::heap::Arbitrary<> heap_type;
    typedef memory::heap::Lock<heap_type, exclusion::Mutex> lockable_heap_type;
    typedef memory::heap::Adapter<lockable_heap_type> adapter_heap_type;

    heap_type               heap;
    lockable_heap_type      lockable_heap(heap);
    adapter_heap_type       adapter_heap(lockable_heap);

    if (!heap.reset(g_buffer, g_buffer_size)) {
        std::cerr << "initialization of global heap failed." << std::endl;
        return -1;
    }
    
    {
        memory::global::Push const push_global_heap(&adapter_heap);

        BOOST_ASSERT(1 < argc);

        std::string const testname = std::string(argv[1]);

        // instantiates writer to output log into a file
        log::writer::File log_writer;

        {
            filesystem::path_type const home = filesystem::home();

            if (home.empty()) {
                hryky_log_emerg("failed to get home directory.");
                return 1;
            }

            namespace fs = boost::filesystem;

            filesystem::path_type app_root(
                home/"kemonomichi.jp"/"hryky-codebase"/("test_" + testname));

            if (!fs::exists(app_root)) {
                fs::create_directories(app_root);
            }

            filesystem::path_type const path =
                app_root / ("test_" + testname + "_log.txt");

            log_writer.path(log::writer::File::path_type(
                path.string<std::string>().c_str()));
        }

        // enregisters the writer here and unregisters the writer at the end of this scope.
        log::writer::Push const push_log_writer(log_writer);

        // retrieves the associative array of unit tests as singleton.
        testing::Registry & registry = testing::Registry::instance();

        // retrieves the corresponding unit test by name.
        testing::Registry::unit_type const & unit = registry[testname.c_str()];

        if (HRYKY_NULLPTR == unit) {
            // output the information as the "emergency" level
            hryky_log_emerg(
                (*reduction::map("invalid testname"))
                 .append("testname", testname.c_str()));
            return 1;
        }

        // invokes the unit test.
        if (!unit->run()) {
            // output the information as the "alert" level
            hryky_log_alert(
                (*reduction::map("failed"))
                .append("testname", testname.c_str()));
            return 1;
        }
    }

    hryky_log_debug(hryky_reduce("heap", heap));

    return 0;
}
