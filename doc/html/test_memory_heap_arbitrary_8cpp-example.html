    // enregisters a test
    class Test : public testing::unit::Base
    {
    public :
        Test() : testing::unit::Base("memory_heap_arbitrary") {}

    private :
        // calls a test.
        virtual bool run_impl();
    };
    Test const g_test;

    // tests memory::heap::Arbitrary with a seed of randomizer
    template < typename OffsetT = memory::heap::Arbitrary<>::offset_type >
    class TestHeap
    {
    public :
        TestHeap(Test & test, uint32_t seed) : test_(test), seed_(seed) {}

        bool run();

    private :
        Test &      test_;
        uint32_t    seed_;
    };

}

/*
  @brief        calls a test.
  @since        2011-12-17T19:23:27+0900
 */
bool Test::run_impl()
{
    uint32_t const seed = static_cast<uint32_t>(std::time(0));

    if (!TestHeap<uint8_t>(*this, seed).run()) {
        hryky_log_emerg("Testing memory::heap::Arbitrary<uint8_t> failed");
        return false;
    }

    if (!TestHeap<uint16_t>(*this, seed).run()) {
        hryky_log_emerg("Testing memory::heap::Arbitrary<uint16_t> failed");
        return false;
    }

    if (!TestHeap<uint32_t>(*this, seed).run()) {
        hryky_log_emerg("Testing memory::heap::Arbitrary<uint32_t> failed");
        return false;
    }

    if (!TestHeap<uint64_t>(*this, seed).run()) {
        hryky_log_emerg("Testing memory::heap::Arbitrary<uint64_t> failed");
        return false;
    }

    if (!TestHeap<>(*this, seed).run()) {
        hryky_log_emerg("Testing memory::heap::Arbitrary<> failed");
        return false;
    }

    return true;
}
/*
  @brief        tests memory::heap::Arbitrary with seed of randomizer.
  @since        2011-11-06T22:10:07+0900
 */
template < typename OffsetT >
bool TestHeap<OffsetT>::run()
{
    uint32_t const seed = this->seed_;

    // initializes logging.
    testing::Log log(this->test_);

    // initializes a generator of pseudo random number.
    testing::Random random;
    random.seed(seed);

    log.reduce().append("randomd", random);

    // initializes the generator of random number according to the normal distribution.
    random::distribution::Normal<size_t> allocation_size_dist(
        random.generator(),
        0x1000,         // deviation
        0x400,          // mean
        0x1,            // min
        0x80000);       // max

    // initializes the generator of random number according to the uniform distribution.
    random::distribution::Uniform<uint32_t> allocations_number_dist(
        random.generator(),
        1,              // min
        0x80000);       // max

    alignment_type const alignment = allocations_number_dist() % 5 + 1;

    log.reduce().append("alignment", static_cast<size_t>(alignment));

    size_t const buffer_size = 0x10000 << alignment;

    void * buffer = std::malloc(buffer_size);
    AutoFree free_buffer(buffer);

    if (HRYKY_NULLPTR == buffer) {
        hryky_log_crit(
            (*reduction::map(
                "failed to allocate buffer "
                "for the test of memory::Heap::Arbitrary"))
            .append("buffer_size", buffer_size));
        return false;
    }

    hryky::memset(buffer, -1, buffer_size);

    typedef memory::heap::Arbitrary<OffsetT> memory_heap_type;

    memory_heap_type heap;

    if (!heap.reset(buffer, buffer_size, alignment)) {
        hryky_log_crit(
            (*reduction::map(
                "failed to reset memory::heap::Arbitrary"))
            .append("buffer_size", buffer_size));
        return false;
    }

    log.reduce().append("heap", heap);

    if (!heap.verify()) {
        hryky_log_crit("Verification failed.");
        return false;
    }

    size_t const allocations_size = allocations_number_dist() % 0x100000 + 0x100000;

    log.reduce().append("allocations size", allocations_size);

    typedef std::list< std::pair<void *, size_t> > allocations_type;

    allocations_type allocations;

    {
        size_t allocation_id = 0;
        for (; allocations_size != allocation_id; ++allocation_id) {
            if (allocations.empty() || 0 == (allocations_number_dist() & 1)) {
                size_t const size = allocation_size_dist();

                alignment_type const alignment =
                    static_cast<alignment_type>((allocations_number_dist() % 8 + 1));

                void * allocated = heap.allocate(size, alignment);

                if (HRYKY_NULLPTR == allocated) {
                    (log.reduce().append(
                        (*reduction::map("failed to allocate"))
                        .append("index",       allocation_id)
                        .append("size",         size)
                        .append("alignment",    static_cast<uint32_t>(alignment))
                        .append("heap",         heap)
                        )
                     );
                    break;
                }

                {
                    uint8_t * ptr = reinterpret_cast<uint8_t *>(allocated);
                    uint8_t * const end = ptr + size;

                    for (; end != ptr; ++ptr) {
                        *ptr = static_cast<uint8_t>(reinterpret_cast<size_t>(ptr));
                    }
                }

#if PRINT_ALLOCATION_HISTORY
                (log.reduce()
                 ->append("allocate", memory_heap_type::Address(heap, allocated))
                 .append("fragments", static_cast<size_t>(heap.free_nodes_size()))
                 .append("heap", heap)
                 );
#endif // PRINT_ALLOCATION_HISTORY

                allocations_type::iterator itr = allocations.begin();

                size_t const offset = allocations_number_dist() % (allocations.size() + 1);

                size_t offset_id = 0;
                for (; offset != offset_id; ++offset_id) {
                    ++itr;
                }

                allocations.insert(
                    itr, allocations_type::value_type(allocated, size));
            }
            else {
                allocations_type::iterator itr = allocations.begin();

                size_t const offset = allocations_number_dist() % allocations.size();

                size_t offset_id = 0;
                for (; offset != offset_id; ++offset_id) {
                    ++itr;
                }

#if PRINT_ALLOCATION_HISTORY
                (log.reduce()
                 ->append("deallocate", memory_heap_type::Address(heap, (*itr).first))
                 .append("fragments", static_cast<size_t>(heap.free_nodes_size()))
                 );
#endif // PRINT_ALLOCATION_HISTORY

                if (!heap.deallocate((*itr).first)) {
                    hryky_log_crit(
                        (*reduction::map("failed to deallocate"))
                        .append("index", allocation_id)
                        .append("address", (*itr).first)
                        .append("size", (*itr).second)
                        .append("heap", heap));
                    return false;
                }

#if PRINT_ALLOCATION_HISTORY
                log.reduce().append("heap", heap);
#endif

                allocations.erase(itr);
            }
        }

        if (!heap.verify()) {
            hryky_log_crit("Verification failed.");
            return false;
        }
    }

    size_t count = 0;

    while (!allocations.empty()) {
        allocations_type::iterator itr = allocations.begin();

        {
            uint8_t * ptr = reinterpret_cast<uint8_t *>((*itr).first);
            uint8_t * const end = ptr + (*itr).second;

            for (; end != ptr; ++ptr) {
                if ((*ptr) !=
                     static_cast<uint8_t>(reinterpret_cast<size_t>(ptr)))
                {
                    hryky_log_crit(
                        (*reduction::map("memory corruption occurred :"))
                        .append("count", count)
                        .append("address", reinterpret_cast<size_t>(ptr))
                        .append("begin", (*itr).first)
                        .append("end", reinterpret_cast<size_t>(end))
                        .append("info", memory_heap_type::Address(heap, (*itr).first))
                        );
                    return false;
                }
            }
        }

        size_t const offset = allocations_number_dist() % allocations.size();

        size_t offset_id = 0;
        for (; offset != offset_id; ++offset_id) {
            ++itr;
        }

#if PRINT_ALLOCATION_HISTORY
        (log.reduce()
         ->append("deallocate", memory_heap_type::Address(heap, (*itr).first))
         .append("fragments", static_cast<size_t>(heap.free_nodes_size()))
         );
#endif // PRINT_ALLOCATION_HISTORY

        if (!heap.deallocate((*itr).first)) {
            hryky_log_crit(
                (*reduction::map("failed to deallocate"))
                 .append("index", count)
                 .append("address", (*itr).first)
                 .append("size", (*itr).second)
                 .append("heap", heap));
            return false;
        }

        allocations.erase(itr);
        ++count;
    }

    if (!heap.verify()) {
        hryky_log_crit("Verification failed.");
        return false;
    }

    // log_out.disable();

    return true;
}
