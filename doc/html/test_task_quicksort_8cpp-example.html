    // enregister a test
    class Test : testing::unit::Base
    {
    public :
        Test() : testing::unit::Base("task_quicksort") {}

    private :
        // tests task::distributor::Batch by using quicksort concurrently.
        virtual bool run_impl();

    };
    Test const g_test;
    
    typedef uint32_t sample_type;

    // a task to print "hello ${this->id_}".
    struct SortTask
    {
        typedef task::distributor::Batch<SortTask>  distributor_type;
        typedef task::worker::Batch<SortTask>       worker_type;

        enum
        {
            // the lower number to create a new task recursively.
            distributable_distance_ = 0x2,
        };
        
        SortTask() : begin_(HRYKY_NULLPTR), end_(HRYKY_NULLPTR) {}
        
        SortTask(sample_type * begin, sample_type * end)
            : begin_(begin), end_(end) {}

        // sorts fixed array concurrently
        void operator()(worker_type & worker);

        reduction_type reduce(reduction::name_type const & name) const;

        sample_type *   begin_;
        sample_type *   end_;
    };

    reduction_type reduce(
        reduction::name_type const & name, SortTask const & src);

    // typedef function_pointer task_type;
    typedef SortTask                        task_type;
    typedef task_type::worker_type          worker_type;
    typedef task_type::distributor_type     distributor_type;

} // namespace *anonymous*

/*
  @brief        tests hryky::task::distributor::Batch by using quicksort concurrently.
  @since        2011-12-17T19:58:12+0900
 */
bool Test::run_impl()
{
    // initializes logging.
    testing::Log log(*this);

    // initializes a random number generator
    uint32_t const          seed = static_cast<uint32_t>(std::time(0));
    random::generator_type  generator;
    generator.seed(seed);

    log.reduce().append("seed", seed);

    // temporary buffer for dynamic memory allocation
    size_t const    buffer_size = 0x400 * 0x400;
    size_t          buffer[buffer_size / sizeof(size_t)] = {0, };
    
    // initializes heap.
    typedef memory::heap::Arbitrary<>   heap_type;
    heap_type                           heap;
    if (!heap.reset(buffer, buffer_size)) {
        hryky_log_alert(
            (*reduction::map(
                "failed to reset memory::heap::Arbitrary"))
            .append("buffer_size",      buffer_size));
        return false;
    }

    // decorates heap as thread-safe.
    typedef memory::heap::Lock<heap_type>   lockable_heap_type;
    lockable_heap_type                      lockable_heap(heap);

    // creates adapter to treat the heap polymorphically.
    typedef memory::heap::Adapter<lockable_heap_type>   adapter_heap_type;
    adapter_heap_type                                   adapter_heap(lockable_heap);

    random::distribution::Normal<size_t> samples_size_dist(
        generator,
        0x1000,     // deviation
        0x10000,    // mean
        1,          // min
        0x100000);  // max
    size_t const samples_size = samples_size_dist();

    (log.reduce()
     .append("samples_size",            samples_size)
     .append("samples_size_dist",       samples_size_dist));

    typedef ScopedArray<sample_type> samples_type;
    samples_type unsorted(new sample_type[samples_size], samples_size);

    if (unsorted.invalid()) {
        hryky_log_emerg(
            "failed to allocate the buffer of samples.");
        return false;
    }

    ScopedArray<sample_type> answer(new sample_type[samples_size], samples_size);

    if (answer.invalid()) {
        hryky_log_emerg(
            "failed to allocate the buffer of samples.");
        return false;
    }

    // makes up sample values
    {
        // pseudo random number uniformly distributed
        random::distribution::Uniform<uint32_t>
            sample_dist(generator, 0, MaxOf<uint32_t>::value);

        Foreach<samples_type> foreach(unsorted);
        for (; foreach.valid(); ++foreach) {
            *foreach = sample_dist();
        }
    }

#if PRINT_VERBOSELY
    {
        reduction::array_type format = reduction::array("unsorted");
        log->append(unsorted);

        Foreach<samples_type> foreach(unsorted);
        for (; foreach.valid(); ++foreach) {
            format->append(*foreach);
        }
    }
#endif // PRINT_VERBOSELY

    memcpy(&*answer.begin(), &*unsorted.begin(), sizeof(sample_type) * samples_size);

    // sorts the samples by this individual context
    clock::duration_type qsort_duration;
    {
        clock::AutoElapsed elapsed(qsort_duration);

        hryky::qsort(answer.begin(), answer.begin() + samples_size);
    }

#if PRINT_VERBOSELY
    {
        reduction::array_type format = reduction::array("answer");
        log->append(format);

        Foreach<samples_type> foreach(answer);
        for (; foreach.valid(); ++foreach) {
            format->append(*foreach);
        }
    }
#endif // PRINT_VERBOSELY

    // checks the sorted result
    {
        sample_type left = 0;

        Foreach<samples_type> foreach(answer);
        for (; foreach.valid(); ++foreach) {
            if (left > *foreach) {
                hryky_log_err(
                    (*reduction::map("quick sort failed"))
                    .append("index", foreach.offset()));
                return false;
            }
            left = *foreach;
        }
    }

    ScopedArray<sample_type> parallel(new sample_type[samples_size], samples_size);

    if (parallel.invalid()) {
        hryky_log_emerg(
            "failed to allocate the buffer of samples.");
        return false;
    }
    memcpy(&*parallel.begin(), &*unsorted.begin(), sizeof(sample_type) * samples_size);

    // initializes a pool of thread resources.
    thread::Pool thread_pool;
    if (!thread_pool.reset(thread::Pool::ResetParameter())) {
        hryky_log_err(
            "failed to reset thread::Pool");
        return false;
    }

    // decides the number of concurrent units
    size_t const sequences_size = 8;

    processor_id_type const preferred_processors[sequences_size] =
    {
        0, 1, 2, 3, 4, 5, 6, 7,
    };

    // initializes the trait of concurrency
    task::Concurrency concurrency;
    if (!concurrency.reset(
        task::Concurrency::ResetParameter(adapter_heap,thread_pool)
        .sequences_size(sequences_size)
        .preferred_processors(preferred_processors)))
    {
        hryky_log_emerg(
            (*reduction::map("failed to reset concurrent module"))
            .append("the number of sequences", sequences_size));
        return false;
    }

    random::distribution::Normal<size_t> batch_size_dist(
            generator,
            0x10 >> 1,  // deviation
            0x10,       // mean
            1,          // min
            0x10 << 4); // max
    size_t const batch_size = batch_size_dist();

    (log.reduce()
     .append("batch_size",          batch_size)
     .append("batch_size_dist",     batch_size_dist));

    // initializes a distributor of task.
    distributor_type distributor;
    if (!distributor.reset(
        distributor_type::ResetParameter(concurrency)
        .batch_size(batch_size)))
    {
        hryky_log_emerg(
            (*reduction::map("failed to reset a task distributor"))
            .append("the size of a batch of tasks", batch_size));
        return false;
    }

    // sorts the samples by concurrent units
    clock::duration_type concurrent_qsort_duration;
    {
        clock::AutoElapsed elapsed(concurrent_qsort_duration);

        if (!distributor.distribute(
            task_type(&*parallel.begin(), &*parallel.end())))
        {
            hryky_log_alert(
                "failed to distribute a task");
            return false;
        }

        // attaches tasks to concurrent units(threads).
        if (!distributor.attach()) {
            hryky_log_alert(
                "failed to attach workers to concurrent units");
            return false;
        }

        // waits to complete the execution of the tasks.
        if (!distributor.join()) {
            hryky_log_alert(
                "failed to wait that the workers complete tasks");
            return false;
        }
    }

#if PRINT_VERBOSELY
    {
        reduction::array_type map = reduction::array("parallel");
        log->append(map);

        Foreach<samples_type> foreach(parallel);
        for (; foreach.valid(); ++foreach) {
            map->append(*foreach);
        }
    }
#endif // PRINT_VERBOSELY

    // checks the sorted result
    {
        sample_type left = 0;

        Foreach<samples_type> foreach(parallel);
        for (; foreach.valid(); ++foreach) {
            if (left > *foreach) {
                hryky_log_err(
                    (*reduction::map("concurrent quick-sort failed"))
                    .append("index",    foreach.offset())
                    .append("lhs",      left)
                    .append("rhs",      *foreach));

                reduction::array_type map = reduction::array("parallel");
                log.reduce().append(map);

                Foreach<samples_type> foreach(parallel);
                for (; foreach.valid(); ++foreach) {
                    map->append(*foreach);
                }

                return false;
            }
            left = *foreach;
        }
    }

    (log.reduce()
     .append("distributor",                     distributor)
     .append("qsort_duration",             qsort_duration.count())
     .append("concurrent_qsort_duration",  concurrent_qsort_duration.count())
     .append("concurrent ratio",
             (static_cast<double>(qsort_duration.count())
              / concurrent_qsort_duration.count())));

    return true;
}
/*
  @brief        sorts fixed array concurrently.
  @since        2011-12-07T00:12:33+0900
 */
void SortTask::operator()(worker_type & worker)
{
    sample_type * const boundary =
        hryky::qsort_internal::partition(this->begin_, this->end_);

    if (boundary == this->end_) {
        return;
    }

    distributor_type * const distributor = worker.distributor();

    size_t const first_distance = boundary - this->begin_;
    size_t const second_distance = this->end_ - boundary;

    if (SortTask::distributable_distance_ > first_distance) {
        if (SortTask::distributable_distance_ > second_distance) {
            // sorts the last half of the region in this thread.
            hryky::qsort(boundary, this->end_);
        }
        else {
            // sorts the last half of the region concurrently.
            if (!distributor->attach_with(SortTask(boundary, this->end_))) {
                hryky_log_alert(
                    reduction::map(
                        "failed to execute a task to sort the last half of the region")
                    ->append("boundary", boundary));
                worker.error();
                return;
            }
        }
        // sorts the first half of the region in this thread.
        hryky::qsort(this->begin_, boundary);
    }
    else if (SortTask::distributable_distance_ > second_distance) {
        // sorts the first half of the region concurrently.
        if (!distributor->attach_with(SortTask(this->begin_, boundary))) {
            hryky_log_alert(
                reduction::map(
                    "failed to execute a task to sort the first half of the region")
                ->append("boundary", boundary));
            worker.error();
            return;
        }

        // sorts the last half of the region in this thread.
        hryky::qsort(boundary, this->end_);
    }
    else if (first_distance > second_distance) {
        // sorts the first half of the region concurrently.
        if (!distributor->attach_with(SortTask(this->begin_, boundary))) {
            hryky_log_alert(
                reduction::map(
                    "failed to execute a task to sort the first half of the region")
                ->append("boundary", boundary));
            worker.error();
            return;
        }

        // sorts the last half of the region in this thread.
        hryky::qsort(boundary, this->end_);
    }
    else {
        // sorts the last half of the region concurrently.
        if (!distributor->attach_with(SortTask(boundary, this->end_))) {
            hryky_log_alert(
                reduction::map(
                    "failed to execute a task to sort the last half of the region")
                ->append("boundary", boundary));
            worker.error();
            return;
        }

        // sorts the first half of the region in this thread.
        hryky::qsort(this->begin_, boundary);
    }

    return;
}
reduction_type SortTask::reduce(reduction::name_type const & name) const {
    reduction::map_type map = reduction::map(name);
    ((*map)
     .append("begin",   this->begin_)
     .append("end",     this->end_)
     );
    return map;
}
reduction_type reduce(
    reduction::name_type const & name, SortTask const & src) {
    return src.reduce(name);
}
