hryky::memory::heap::Arbitrary

hryky::memory::heap::Arbitrary allocates memory space from passed buffer. The size of the allocating region can be arbitrary. Besides, the address alignment can be specified.

Following example is based on test_memory_heap_arbitrary.cpp

Initialization

    typedef memory::heap::Arbitrary<OffsetT> memory_heap_type;

    memory_heap_type heap;

    if (!heap.reset(buffer, buffer_size, alignment)) {
        hryky_log_crit(
            (*reduction::map(
                "failed to reset memory::heap::Arbitrary"))
            .append("buffer_size", buffer_size));
        return false;
    }

allocations of fixed size memory

                void * allocated = heap.allocate(size, alignment);

                if (HRYKY_NULLPTR == allocated) {
                    (log.reduce().append(
                        (*reduction::map("failed to allocate"))
                        .append("index",       allocation_id)
                        .append("size",         size)
                        .append("alignment",    static_cast<uint32_t>(alignment))
                        .append("heap",         heap)
                        )
                     );
                    break;
                }

deallocating the memory

                if (!heap.deallocate((*itr).first)) {
                    hryky_log_crit(
                        (*reduction::map("failed to deallocate"))
                        .append("index", allocation_id)
                        .append("address", (*itr).first)
                        .append("size", (*itr).second)
                        .append("heap", heap));
                    return false;
                }

hryky::memory::heap::Fixed

hryky::memory::heap::Fixed allocates a memory block from passed buffer. The size of the allocating region is fixed length. The address alignment is hryky::memory::heap::Fixed::min_alignment_ bits.

Following example is based on test_memory_heap_fixed.cpp

Initialization

    typedef hryky::memory::heap::Fixed memory_heap_type;
    memory_heap_type heap;

    if (!heap.reset(buffer, buffer_size, unit_size)) {
        hryky_log_crit(
            (*reduction::map("failed to reset memory::heap::Arbitrary"))
            .append("buffer_size", buffer_size)
            );
        return false;
    }

allocations of fixed size memory

                void * allocated = heap.allocate();

                if (HRYKY_NULLPTR == allocated) {
                    (log.reduce()
                     .append(
                         (*reduction::map("failed to allocate"))
                         .append("index", allocation_id)
                         .append("heap", heap)
                         )
                     );

                    if (0 != heap.available_size()) {
                        hryky_log_crit(
                            (*reduction::map("unexpected allocation failure occurred"))
                            .append("allocation_id", allocation_id)
                            .append("units_size", units_size)
                            );
                        return false;
                    }
                    
                    break;
                }

deallocating the memory

                if (!heap.deallocate(*itr)) {
                    hryky_log_crit(
                        (*reduction::map("failed to deallocate"))
                        .append("index",    allocation_id)
                        .append("address",  *itr)
                        .append("heap",     heap)
                        );
                    return false;
                }
