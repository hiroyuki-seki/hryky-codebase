/*
  @brief        tests hryky::memory::heap::Fixed.
  @since        2011-12-17T19:45:38+0900
 */
bool Test::run_impl()
{
    // initialize logging.
    testing::Log log(*this);

    // initialize a generator of pseudo random number.
    testing::Random random;

    log.reduce().append("random", random);
    
    // initializes the generator of pseudo-random number according to the uniform distribution.
    random::distribution::Uniform<uint32_t>
        allocations_number_dist(random.generator(), 1, 0x1000);

    typedef hryky::memory::heap::Fixed memory_heap_type;

    // initializes the generator of pseudo-random number according to the normal distribution.
    random::distribution::Normal<memory_heap_type::index_type> units_size_dist(
        random.generator(),
        0x1000,         // deviation
        0x400,          // mean
        1,              // min
        0x10000);       // max
    memory_heap_type::index_type const units_size = units_size_dist();

    // initializes the generator of pseudo-random number according to the normal distribution.
    random::distribution::Normal<memory_heap_type::index_type> unit_size_dist(
        random.generator(),
        0x40,           // deviation
        0x40,           // mean
        1,              // min
        0x10000);       // max
    size_t const unit_size = unit_size_dist();

    alignment_type const alignment = memory_heap_type::min_alignment_;

    size_t const buffer_size =
        memory_heap_type::buffer_size(unit_size, units_size);

    (log.reduce()
     .append("unit_size", unit_size)
     .append("units_size", units_size)
     .append("buffer_size", buffer_size)
     );

    void * buffer = std::malloc(buffer_size + (1 << alignment));
    AutoFree free_buffer(buffer);

    if (HRYKY_NULLPTR == buffer) {
        hryky_log_crit(
            (*reduction::map("failed to allocate buffer for the test of memory::Heap::Arbitrary"))
            .append("the size of one unit", unit_size)
            .append("the number of units",  units_size)
            .append("the size of buffer",   buffer_size)
            );
        return false;
    }

    hryky::memset(buffer, -1, buffer_size);

    memory_heap_type heap;

    if (!heap.reset(buffer, buffer_size, unit_size)) {
        hryky_log_crit(
            (*reduction::map("failed to reset memory::heap::Arbitrary"))
            .append("buffer_size", buffer_size)
            );
        return false;
    }

    if (!heap.verify()) {
        hryky_log_crit("Verification failed.");
        return false;
    }

    size_t const allocations_size_width =   0x10000;
    size_t const allocations_size_min =     0x1000;

    size_t const allocations_size =
        allocations_number_dist() % allocations_size_width + allocations_size_min;

    log.reduce().append("allocations size", allocations_size);

    typedef std::list< void * > allocations_type;

    allocations_type allocations;

    {
        size_t allocation_id = 0;
        for (; allocations_size != allocation_id; ++allocation_id) {
            if (allocations.empty() || 0 == (allocations_number_dist() & 1)) {
                void * allocated = heap.allocate();

                if (HRYKY_NULLPTR == allocated) {
                    (log.reduce()
                     .append(
                         (*reduction::map("failed to allocate"))
                         .append("index", allocation_id)
                         .append("heap", heap)
                         )
                     );

                    if (0 != heap.available_size()) {
                        hryky_log_crit(
                            (*reduction::map("unexpected allocation failure occurred"))
                            .append("allocation_id", allocation_id)
                            .append("units_size", units_size)
                            );
                        return false;
                    }
                    
                    break;
                }

                {
                    uint8_t * ptr = reinterpret_cast<uint8_t *>(allocated);
                    uint8_t * const end = ptr + unit_size;

                    for (; end != ptr; ++ptr) {
                        *ptr = static_cast<uint8_t>(reinterpret_cast<size_t>(ptr));
                    }
                }

#if PRINT_ALLOCATION_HISTORY
                (log.reduce()
                 .append("allocate",    memory_heap_type::Address(heap, allocated))
                 .append("available",   heap.available_size())
                 );
#endif // PRINT_ALLOCATION_HISTORY

                allocations_type::iterator itr = allocations.begin();

                size_t const offset = allocations_number_dist() % (allocations.size() + 1);

                size_t offset_id = 0;
                for (; offset != offset_id; ++offset_id) {
                    ++itr;
                }

                allocations.insert(
                    itr, allocations_type::value_type(allocated));
            }
            else {
                allocations_type::iterator itr = allocations.begin();

                size_t const offset = allocations_number_dist() % allocations.size();

                size_t offset_id = 0;
                for (; offset != offset_id; ++offset_id) {
                    ++itr;
                }

#if PRINT_ALLOCATION_HISTORY
                (log.reduce()
                 .append("deallocate",  memory_heap_type::Address(heap, *itr))
                 .append("avaiable",    heap.available_size())
                 );
#endif // PRINT_ALLOCATION_HISTORY

                if (!heap.deallocate(*itr)) {
                    hryky_log_crit(
                        (*reduction::map("failed to deallocate"))
                        .append("index",    allocation_id)
                        .append("address",  *itr)
                        .append("heap",     heap)
                        );
                    return false;
                }

                allocations.erase(itr);
            }
        }

        if (!heap.verify()) {
            hryky_log_crit("Verification failed.");
            return false;
        }
    }

    size_t count = 0;

    while (!allocations.empty()) {
        allocations_type::iterator itr = allocations.begin();

        {
            uint8_t * ptr = reinterpret_cast<uint8_t *>(*itr);
            uint8_t * const end = ptr + unit_size;

            for (; end != ptr; ++ptr) {
                if ((*ptr) !=
                     static_cast<uint8_t>(reinterpret_cast<size_t>(ptr)))
                {
                    hryky_log_crit(
                        (*reduction::map("memory corruption occurred"))
                        .append("count",                                    count)
                        .append("octet_address",                            ptr)
                        .append("begin",                                    *itr)
                        .append("end",                                      end)
                        .append("address", memory_heap_type::Address(heap,  *itr))
                        );
                    return false;
                }
            }
        }

        size_t const offset = allocations_number_dist() % allocations.size();

        size_t offset_id = 0;
        for (; offset != offset_id; ++offset_id) {
            ++itr;
        }

#if PRINT_ALLOCATION_HISTORY
        (log.reduce()
         .append("deallocate", memory_heap_type::Address(heap, *itr))
         .append("available", heap.available_size())
         );
#endif // PRINT_ALLOCATION_HISTORY

        if (!heap.deallocate(*itr)) {
            hryky_log_crit(
                (*reduction::map("failed to deallocate"))
                .append("index",    count)
                .append("address",  *itr)
                .append("heap",     heap)
                );
            return false;
        }

        allocations.erase(itr);
        ++count;
    }

    if (!heap.verify()) {
        hryky_log_crit("Verification failed.");
        return false;
    }

    // log_out.disable();

    return true;
}
