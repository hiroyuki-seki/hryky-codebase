    // This class can be used as the second argument of hryky::reduction::Map::append().
    class UserValue

    public :

/*
  @brief        tests log modules.
  @since        2011-12-29T20:41:33+0900
 */
bool Test::run_impl()
{
    // prints key-value pair
    hryky_log_debug(
        hryky_reduce("int_value_1",   int(1)));

    // instantiates map
    reduction::map_type map = reduction::map("map");

    // aggregates multiple key-value pairs
    ((*map)
     .append("int_10",          10)
     .append("float_10.0f",     10.0f)
     .append("bool_true",       true));

    // appends another map as a child
    ((*map)
     .append(
         (*reduction::map("aggregation"))
         .append("int_20",          int(20))
         .append("float_20.0f",     float(20.0f))
         .append("bool_true",       bool(true))
         .append("user_value",      UserValue(4))
         )
     );

    // prints formatted hierarchy
    hryky_log_debug(*map);

    hryky_log_debug(
        reduction::map("parent")->append(*map));
    
    return true;
}
} // namespace "anonymous"

/*
  @brief        creates reduction::Base corresponding to UserValue.
  @since        2012-01-03T12:56:28+0900
 */
hryky::reduction_type hryky::test::reduce(
    reduction::name_type const & name, UserValue const & param) {
    // instantiates reduction_type as key-value pairs.
    reduction::map_type map = reduction::map(name);

    // appends the child pair to the reduction::map_type.
    map->append("value", param.value());

    // returns hryky::reduction_type
    return map;
}
